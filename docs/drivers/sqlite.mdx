### Monitoramento de Performance

```javascript
class SQLiteMonitor {
    constructor(driver) {
        this.driver = driver;
        this.metricas = {
            consultas: 0,
            tempoTotal: 0,
            operacoesLentas: []
        };
    }
    
    medirOperacao(operacao, ...args) {
        const inicio = Date.now();
        
        try {
            const resultado = operacao.call(this.driver, ...args);
            const tempo = Date.now() - inicio;
            
            this.metricas.consultas++;
            this.metricas.tempoTotal += tempo;
            
            if (tempo > 100) { // Opera√ß√µes > 100ms
                this.metricas.operacoesLentas.push({
                    tempo,
                    operacao: operacao.name,
                    timestamp: new Date()
                });
            }
            
            return resultado;
        } catch (error) {
            console.error('Erro na opera√ß√£o:', error);
            throw error;
        }
    }
    
    obterEstatisticas() {
        return {
            ...this.metricas,
            tempoMedio: this.metricas.tempoTotal / this.metricas.consultas,
            operacoesLentasRecentes: this.metricas.operacoesLentas.slice(-10)
        };
    }
}
```

## Troubleshooting

### Problemas Comuns
